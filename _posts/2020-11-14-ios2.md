## Operator
```swift

let a = 10
let b = 20

a == b
//false

a != b
//true

if a == b {
	print("같다")
}else{
	print("다르다")
}

a == b ? print("같다") : print("다르다"

let aa = false

!aa
//true


//??
var aa: Int? 
//aa는 int값이 되거나 널값이 될 수 있다.

print(aa ?? 0)
//aa가 빈 값일 경우 0으로 나온다.
//숫자가 무조건 나와야 하는 경우 사용해주면 좋다.

```
